Map

let map = new Map();
map.set('nombre', 'Juan');
map.set('edad', 30);

for (let [clave, valor] of map) {
	console.log(clave + ': ' + valor)
}

Arreglo de objetos

let arreglo = [
	{clave: 'nombre', valor: 'Juan'},
	{clave: 'edad', valor: 30} 
]

for (let elemento of arreglo) {
	console.log(elemento.clave + ': ' + elemento.valor)
}
